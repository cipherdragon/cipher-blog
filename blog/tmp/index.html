START
    DECLARE STRING nic_number
    DECLARE INT nic_length

    DECLARE BOOLEAN valid_voter = TRUE
    DECLARE STRING gender
    DECLARE INT birth_year
    DECLARE INT age
    DECLARE INT birthday_of_year

    GET NIC number as nic_number
    nic_length = length of nic_number

    IF nic_length == 10 THEN
        nic_birth_year = first two characters of nic_number
        nic_birthday_of_year = 3rd to 5th characters (inclusive) of nic_number
        vote_validity_letter = 10th character of nic_number

        birth_year = (convert nic_birth_year to INTEGER) + 1900
        birthday_of_year = convert nic_birthday_of_year to INTEGER


        IF vote_validity_letter == "X" THEN
            valid_voter = FALSE
        END IF

    ELSE IF nic_length == 12 THEN
        nic_birth_year = first four characters of nic_number
        nic_birthday_of_year = 5th to 7th characters (inclusive) of nic_number

        birth_year = convert nic_birth_year to INTEGER
        birthday_of_year = convert nic_birthday_of_year to INTEGER

        IF birthday_of_year > 500 THEN
            gender = "FEMALE"
        ELSE
            gender = "MALE"
        END IF

        age = (current year) - birth_year

    ELSE
        DISPLAY "Entered NIC is not valid"
        STOP
    END IF

    IF birthday_of_year > 500 THEN
        gender = "FEMALE"
    ELSE
        gender = "MALE"
    END IF
    age = (current year) - birth_year

    IF valid_voter != FALSE THEN
        IF age < 18 THEN
            valid_voter = FALSE
        ELSE
            valid_voter = TRUE
        END IF
    END IF

    IF valid_voter == TRUE THEN
        DISPLAY "You are a valid voter"
    ELSE
        DISPLAY "You are an invalid voter"
    END IF

    voter_data_file = Create a file with nic_number as the name    

    APPEND STRING ("nic : ", nic_number) to voter_data_file
    APPEND STRING ("birth_year : ", birth_year) to voter_data_file
    APPEND STRING ("gender : ", gender) to voter_data_file
    APPEND STRING ("is_able_to_vote : ", valid_voter) to voter_data_file
STOP
